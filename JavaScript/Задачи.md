```javascript
/*
Дан массив слов, необходмо найти все слова анаграммы и вернуть массив анаграмм по возрастанию от количества повторений.

В цикле проходим по каждому элементу-слову исходного массива. Кадое слово конвертируем в массив и сортируем его. Далее, каждое слово записываем в словарь, где ключ - это слово, а значение - кол-во вхождений. Затем сортируем по кол-ву вхождений.
*/

const words = ['ear', 'rea', 'clon']

const getAnagramms = (arr) => {
	const wordMap = {}
  
  arr.forEach((el) => {
  	const convertedEl = el
	  	.toLowerCase()
	  	.split('')
	  	.sort()
	  	.join('')
    
    if (wordMap[convertedEl]) {
    	wordMap[convertedEl] += 1 
    } else {
    	wordMap[convertedEl] = 1
    }
  })
  
  return Object
  	.entries(wordMap)
  	.sort((a, b) => a[1] - b[1])
    .map(el => el[0])
}

console.log(getAnagramms(words));
```

/*
	Отсортировать чётные элементы массава в порядке возрастания, оставив их на месте.
*/

const sort = (arr) => {
	const evens = []
  
  arr.forEach((el) => {
  	if (el % 2 === 0) {
    	evens.push(el)
    }
  })
  
  evens.sort((a, b) => a - b)
  
  return arr.map((el) => 
  	el % 2 === 0 ?
    	evens.shift() :
      el
  )
}

console.log(sort([1, 8, 3, 7, 2])); // - [1, 2, 3, 7, 8]

	const numberMap = {}
  
  for (let i = 0; i < arr.length; i++) {
  	if (!numberMap[arr[i]]) {
    	numberMap[arr[i]] = 1

const memo = (fn) => {
	const prevArgs = new Map()

	return (...args) => {
		const strArgs = JSON.stringify(args);

		if (!prevArgs.has(strArgs)) {
			const res = fn(...args)
			prevArgs.set(strArgs, res)

			return res;
		}

		return prevArgs.get(strArgs)
	}
}

Пример реализации Event Emitter

```javascript
class EventEmitter {
  constructor() {
    this.events = {};
  }

  on(event, listener) {
    if (!this.events[event]) {
      this.events[event] = [];
    }
    this.events[event].push(listener);
    return this; // для чейнинга
  }

  off(event, listener) {
    if (!this.events[event]) return this;
    this.events[event] = this.events[event].filter(l => l !== listener);
    return this;
  }

  once(event, listener) {
    const wrapper = (...args) => {
      listener(...args);
      this.off(event, wrapper);
    };
    this.on(event, wrapper);
    return this;
  }

  emit(event, ...args) {
    if (!this.events[event]) return false;
    this.events[event].forEach(listener => listener(...args));
    return true;
  }
}

// --- Пример использования ---
const emitter = new EventEmitter();

function greet(name) {
  console.log(`Привет, ${name}!`);
}

emitter.on('hello', greet);

emitter.emit('hello', 'Алекс');  // Привет, Алекс!
emitter.off('hello', greet);

emitter.emit('hello', 'Алекс');  // уже ничего не выводится

// once
emitter.once('ping', () => console.log('ping!'));
emitter.emit('ping');  // ping!
emitter.emit('ping');  // ничего

```

Генерация случайных чисел с помощью Math.random - возвращает псевдослучайное число
в диапазоне [0, 1)

```javascript
function randomInt(min, max) {
  return Math.floor(Math.random() * (max - min + 1)) + min;
}

randomInt(3, 7); // 3–7
```
