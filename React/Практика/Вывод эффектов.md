    console.log(state, 'Hack Frontend effect');

    return () => {
      console.log(state, 'Hack Frontend effect cleanup');
    };
  }, [state]);

  useLayoutEffect(() => {
    console.log(state, 'Hack Frontend layout effect');

    return () => {
      console.log(state, 'Hack Frontend layout cleanup');
    };
  }, [state]);

  return (
    <div>
      <button onClick={() => setState((prev) => prev + 1)}>
        click (current: {state})
      </button>
    </div>
  );
}
