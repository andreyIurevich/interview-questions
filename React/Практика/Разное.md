```jsx
import { useState } from 'react';

export default function Counter() {
  const [count, setCount] = useState(0);

  const handleClick = () => {
    setCount(count + 1);
    setCount(count + 1);
    
    setTimeout(() => {
      setCount(count + 1);
    }, 1000);
  };

  console.log('Рендер со значением:', count);

  return (
    <button onClick={handleClick}>
      Увеличить. Текущий счет: {count}
    </button>
  );
}
/*
Вывод:
Рендер со значением: 0 (начальное значение)
Рендер со значением: 1 (после клика)
Рендер со значением: 1 (через секунду, из setTimeout)
*/
```

- Переменная `count` из первого рендера и переменная `count` из второго рендера — это **два разных участка в оперативной памяти компьютера**.
- Таймер, запущенный на первом рендере, умеет смотреть только в первый участок памяти.

```jsx
import { useState, memo } from 'react';

// Компонент списка (оптимизирован через memo)
const UserList = memo(({ items, onDelete }) => {
  console.log('--- Рендер списка UserList ---');
  return (
    <ul>
      {items.map(user => (
        <li key={user.id}>
          {user.name} <button onClick={() => onDelete(user.id)}>Удалить</button>
        </li>
      ))}
    </ul>
  );
});

// Главный компонент страницы
export default function App() {
  const [count, setCount] = useState(0);
  const [users, setUsers] = useState([
    { id: 1, name: 'Иван' },
    { id: 2, name: 'Анна' }
  ]);

  const handleDelete = (id) => {
    setUsers(users.filter(user => user.id !== id));
  };

  return (
    <div>
      <button onClick={() => setCount(count + 1)}>
        Кликнуто раз: {count}
      </button>
      
      <UserList 
        items={users} 
        onDelete={handleDelete} 
      />
    </div>
  );
}
```

При клике на кнопку компонент UserList будет перерендериваться, т.к. будет создаваться новая функция handleDelete. Стоит обернуть handleDelete в useCallback, чтобы она не зависела от count:

```jsx
const handleDelete = useCallback((id) => {
  setUsers(prevUsers => prevUsers.filter(user => user.id !== id));
}, []); // Массив зависимостей пустой! Ссылка вечная.
```
