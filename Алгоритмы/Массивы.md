##### Задачи

Вывести дубликаты в массиве

```js
function duble(arr) {
	let obj = {};
	let res = [];
  
  for (let i = 0; i < arr.length; i++) {
  	if (!obj[arr[i]]) {
    	obj[arr[i]] = 1;
    } else {
    	res.push(arr[i]);
    }
  }
  
  return res;
}

console.log(duble([1, 2, 3, 4, 3, 2])); // 2, 3
```
