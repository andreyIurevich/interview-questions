```javascript
const a = { a: 'a' };
const b = { b: 'b' };
const c = {};

// Выполяется приведение к строке ключа
c[a] = a; // [Object Object]: { a: 'a' }
c[b] = b; // перезапись ключа [Object Object]: { b: 'b' }

console.log(c[a].a, c[b].b); // undefined, b
```

```typescript
function sleep<T extends number>(ms: T): Promise<T> {
	return new Promise((resolve) => setTimeout(() => resolve(ms), ms))
}

sleep(1000)
	.then((res) => {
		console.log(res);
	})
	.then((res) => {
		console.log(res);
		return sleep(500);
	})
	.then((res) => {
		console.log(res);
	})

// 1000 undefined 500
```

```javascript
// Проверка скобок - входная строка содержит символы ()

const checkBrackets = (str = '') => {
	let arr = []
  
  for (let i = 0; i < str.length; i++) {
  	if (str[i] === "(") {
    	arr.push(str[i])
    } else if (arr[arr.length - 1] === "(") {
    	arr.pop()
    } else {
    	arr.push(str[i])
    }
  }
  
  return arr.length === 0
}

console.log(checkBrackets('(()()))')); // false
console.log(checkBrackets('(()())')); // true

// Проверка скобок - ([{}])
```
